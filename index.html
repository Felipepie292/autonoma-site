// autonoma.agent.config
import { Agent } from '@autonoma/core'

const agent = new Agent({
  name:     'atendimento-v2',
  model:    'gpt-4-turbo',
  context:  'empresa.dados',
  memory:   true,
  learn:    true,
})

// Processar conversas em tempo real
agent.on('message', async (data) => {
  const insight = await agent.analyze(data)
  const action  = await agent.decide(insight)

  if (action.type === 'respond') {
    return agent.reply(action.content)
  }

  // Escalar para humano quando necessário
  return agent.escalate({
    to:      'equipe.suporte',
    context: insight,
    urgency: action.priority
  })
})

agent.deploy()  // ● online
agent.status: running
